exercise 1

SELECT YEAR('enrolment_date'),
COUNT(*) AS 'registrati'
FROM 'students'
GROUP BY YEAR('enrolment_date');

exercise 2

SELECT 'office_address',
COUNT(*) AS 'teachers'
FROM 'teachers'
GROUP BY YEAR('office_address');

exercise 3

SELECT 'exam_id',
AVG('vote') AS 'voto_medio'
FROM 'exam_students'
GROUP BY YEAR('exam_id');
